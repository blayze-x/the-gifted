function initLevel(i){
  const lv=LEVELS[i];
  player={x:lv.sx,y:lv.sy,w:16,h:32,vx:0,vy:0,onGround:false,facing:1,
    dashCD:0,dashing:false,dashT:0,hp:Math.min((player&&player.hp)||100,100),blastCD:0,
    invincible:0,walkT:0,walkDir:0,capeT:0,glowT:0,jumpBuf:0,shooting:0};
  coins=lv.coins.map(c=>({...c,collected:false,bob:Math.random()*Math.PI*2}));
  crumbleState={};particles=[];magicBolts=[];minions=[];
  trapStates=lv.traps.map(t=>({...t,a:t.a||0,cx:t.x||0,dir:t.dir||1}));
  if(lv.boss){
    boss={hp:lv.bossHp,maxHp:lv.bossHp,x:320,y:320,w:40,h:56,vx:1.8,vy:0,
      onGround:false,shootT:0,jumpT:20,phase:0,angry:false,
      swingT:0,charge:false,chargeT:0};
    // Spawn 5 minions for final boss (level index 14)
    if(i===14){
      minions=[];
      const mxPos=[80,180,300,450,580];
      for(let m=0;m<5;m++){
        minions.push({x:mxPos[m],y:310,w:20,h:28,vx:(m%2===0?1:-1)*1.2,vy:0,
          hp:1,alive:true,onGround:false,shootT:60+m*20,facing:1});
      }
    }
    document.getElementById('bossbar').style.display='block';
    document.getElementById('bossname').textContent=lv.bossName;
  } else {
    boss=null;
    document.getElementById('bossbar').style.display='none';
  }
  document.getElementById('stageName').textContent=lv.name;
  if(musicStarted){ if(lv.boss) startMusic('boss'); else startMusic(cLvl>=10?'void':cLvl>=5?'forest':'dungeon'); }
  updateHUD();
}

// ── PARTICLES ─────────────────────────────────────────────────────────────────
function addP(x,y,col,vx,vy,life,sz=3){
  if(particles.length>40) return;
  particles.push({x,y,col,vx,vy,life,maxLife:life,sz});
}
function burst(x,y,col,n=6,spd=4,life=20,sz=3){
  for(let i=0;i<n;i++) addP(x,y,col,(Math.random()-0.5)*spd*2,(Math.random()-0.5)*spd*2,life+(Math.random()*10|0),sz);
}

// ── HURT ─────────────────────────────────────────────────────────────────────
function hurtPlayer(dmg){
  if(player.invincible>0)return;
  player.hp-=dmg;player.invincible=70;player.vy=-4;
  burst(player.x+8,player.y+16,'#f87171',8,5,22);
  sfxHurt();
  if(player.hp<=0){
    player.hp=0;died=true;
    showMsg('<span class="title">YOU DIED</span>Score: '+score+'<span class="sub">Press any key to retry</span>',9999);
  }
}

// ── TRAPS ────────────────────────────────────────────────────────────────────
function updateTraps(){
  const lv=LEVELS[cLvl];
  for(const t of trapStates){
    if(t.type==='pendulum'){
      t.a+=t.spd;
      const bx=t.x+Math.sin(t.a)*t.len,by=t.y+Math.cos(t.a)*t.len;
      if(overlap(player.x,player.y,player.w,player.h,bx-9,by-9,18,18)) hurtPlayer(20);
    }
    if(t.type==='fireball'||t.type==='saw'){
      t.cx+=t.dir*(t.vx||2.5);
      if(t.cx>t.startX+t.range||t.cx<t.startX) t.dir*=-1;
      if(overlap(player.x,player.y,player.w,player.h,t.cx-9,t.y-9,18,18)) hurtPlayer(t.type==='saw'?25:15);
    }
    if(t.type==='spike'){
      if(player.x+player.w>t.x&&player.x<t.x+t.w&&player.y+player.h>=408&&player.y+player.h<=420) hurtPlayer(30);
    }
  }
}

function overlap(ax,ay,aw,ah,bx,by,bw,bh){
  return ax<bx+bw&&ax+aw>bx&&ay<by+bh&&ay+ah>by;
}

// ── MINIONS ──────────────────────────────────────────────────────────────────
function updateMinions(){
  if(!minions.length) return;
  for(let i=minions.length-1;i>=0;i--){
    const m=minions[i];
    if(!m.alive) continue;
    // Patrol back and forth
    m.x+=m.vx;
    m.facing=m.vx>0?1:-1;
    m.vy+=0.47;
    m.y+=m.vy;
    // Platform collision
    m.onGround=false;
    for(const p of LEVELS[cLvl].plats){
      if(m.x+m.w>p.x&&m.x<p.x+p.w&&m.y+m.h>p.y&&m.y+m.h<p.y+p.h+10&&m.vy>=0){
        m.y=p.y-m.h; m.vy=0; m.onGround=true;
      }
    }
    // Bounce off walls
    if(m.x<0){m.x=0;m.vx*=-1;}
    if(m.x+m.w>W){m.x=W-m.w;m.vx*=-1;}
    if(m.y>H+60){m.y=300;m.vy=0;}
    // Shoot at player every ~90 frames
    m.shootT--;
    if(m.shootT<=0){
      m.shootT=80+Math.random()*40|0;
      const dx=(player.x+8)-(m.x+10),dy=(player.y+16)-(m.y+14);
      const dist=Math.sqrt(dx*dx+dy*dy)||1;
      if(magicBolts.length<20) magicBolts.push({x:m.x+10,y:m.y+10,vx:dx/dist*3,vy:dy/dist*3,fromBoss:true,life:160});
    }
    // Hurt player on contact
    if(overlap(player.x,player.y,player.w,player.h,m.x,m.y,m.w,m.h)) hurtPlayer(10);
    // Check if hit by player bolt
    for(let b=magicBolts.length-1;b>=0;b--){
      const bolt=magicBolts[b];
      if(!bolt.fromBoss&&overlap(bolt.x-5,bolt.y-5,10,10,m.x,m.y,m.w,m.h)){
        m.alive=false;
        burst(m.x+10,m.y+14,'#a78bfa',10,5,25);
        burst(m.x+10,m.y+14,'#fbbf24',6,3,18);
        score+=30;
        magicBolts.splice(b,1);
        break;
      }
    }
  }
  // Remove dead minions
  for(let i=minions.length-1;i>=0;i--){
    if(!minions[i].alive) minions.splice(i,1);
  }
}

function updateBoss(){
  if(!boss||boss.hp<=0) return;
  boss.angry=boss.hp<boss.maxHp*0.5;
  const spd=boss.angry?3.2:2.0;
  boss.swingT+=boss.angry?0.05:0.035;

  if(!boss.charge){
    boss.vx=Math.sign((player.x+8)-(boss.x+20))*spd;
  } else {
    boss.vx=boss.chargeDir*9;
    boss.chargeT--;
    if(boss.chargeT<=0) boss.charge=false;
  }
  boss.vy+=0.48;
  boss.x+=boss.vx; boss.y+=boss.vy;
  boss.x=Math.max(0,Math.min(W-boss.w,boss.x));

  boss.onGround=false;
  for(const p of LEVELS[cLvl].plats){
    if(boss.x+boss.w>p.x&&boss.x<p.x+p.w&&boss.y+boss.h>p.y&&boss.y+boss.h<p.y+p.h+12&&boss.vy>=0){
      boss.y=p.y-boss.h; boss.vy=0; boss.onGround=true;
    }
  }
  if(boss.onGround){
    boss.jumpT--;
    if(boss.jumpT<=0){boss.vy=-14+(boss.angry?-3:0);boss.jumpT=55+Math.random()*40|0;}
    if(boss.angry&&Math.random()<0.025&&!boss.charge){
      boss.charge=true;boss.chargeT=22;boss.chargeDir=Math.sign((player.x+8)-(boss.x+20));
      burst(boss.x+20,boss.y+30,'#ef4444',12,5,20);
    }
  }
  boss.shootT--;
  if(boss.shootT<=0){
    boss.shootT=boss.angry?30:50;
    const shots=boss.angry?5:2;
    for(let s=0;s<shots;s++){
      const spread=(s-(shots-1)/2)*0.25;
      const dx=(player.x+8)-(boss.x+20),dy=(player.y+16)-(boss.y+28);
      const dist=Math.sqrt(dx*dx+dy*dy)||1;
      const spd2=boss.angry?5.5:4.2;
      if(magicBolts.length<20) magicBolts.push({x:boss.x+20,y:boss.y+28,vx:dx/dist*spd2+spread,vy:dy/dist*spd2,fromBoss:true,life:200});
    }
  }
  if(overlap(player.x,player.y,player.w,player.h,boss.x,boss.y,boss.w,boss.h)) hurtPlayer(18);
  if(boss.y>H+80){boss.y=50;boss.vy=0;}
}

// ── MAGIC BOLTS ──────────────────────────────────────────────────────────────
function updateBolts(){
  for(let i=magicBolts.length-1;i>=0;i--){
    const b=magicBolts[i];
    b.x+=b.vx; b.y+=b.vy; b.life--;
    if(gFrame%8===0) addP(b.x,b.y,b.fromBoss?'#f87171':'#c4b5fd',0,0,4,2);
    if(b.life<=0){magicBolts.splice(i,1);continue;}
    if(!b.fromBoss&&boss&&boss.hp>0&&overlap(b.x-5,b.y-5,10,10,boss.x,boss.y,boss.w,boss.h)){
      boss.hp-=25; burst(b.x,b.y,'#f87171',10,5,22); burst(b.x,b.y,'#fbbf24',6,3,18);
      magicBolts.splice(i,1);
      if(boss.hp<=0){
        burst(boss.x+20,boss.y+28,'#fbbf24',8,6,30,4);
        burst(boss.x+20,boss.y+28,'#a78bfa',6,5,28,4);
        sfxBossDie();
        score+=300;
        levelLocked=true;
        const nextAfterBoss=cLvl+1;
        showMsg('<span class="title">BOSS DEFEATED!</span>+300 bonus!<span class="sub">Loading next level...</span>',130);
        setTimeout(()=>{if(nextAfterBoss<=14){cLvl=nextAfterBoss;initLevel(cLvl);levelLocked=false;autoSave();}else{showMsg('<span class="title">YOU WIN!</span>Final Score: '+score+'<br>The Gifted has triumphed!',9999);}},2200);
      }
      continue;
    }
    if(b.fromBoss&&overlap(b.x-5,b.y-5,10,10,player.x,player.y,player.w,player.h)){
      hurtPlayer(18); magicBolts.splice(i,1);
    }
  }
}

// ── PLAYER ───────────────────────────────────────────────────────────────────
let lastJump=false,lastZ=false,lastSh=false;
function updatePlayer(){
  const jP=keys['ArrowUp']||keys['w']||keys['W']||keys[' '];
  const zP=keys['z']||keys['Z'];
  const shP=keys['Shift'];
  const lP=keys['ArrowLeft']||keys['a']||keys['A'];
  const rP=keys['ArrowRight']||keys['d']||keys['D'];

  player.capeT+=0.07; player.glowT+=0.04;
  player.dashCD=Math.max(0,player.dashCD-1);
  player.invincible=Math.max(0,player.invincible-1);
  player.shooting=Math.max(0,player.shooting-1);

  if(player.dashing){
    player.dashT--;
    player.vx=player.facing*7.0;
    if(gFrame%2===0) addP(player.x+8,player.y+16,'#38bdf8',-player.facing*(1+Math.random()*2),(Math.random()-0.5)*1.5,14,3);
    if(player.dashT<=0){player.dashing=false;player.vx=0;player.walkDir=0;}
  } else {
    if(lP){player.vx=-2.2;player.facing=-1;player.walkT+=0.18;player.walkDir=-1;}
    else if(rP){player.vx=2.2;player.facing=1;player.walkT+=0.18;player.walkDir=1;}
    else{player.vx*=0.68;if(Math.abs(player.vx)<0.05)player.vx=0;player.walkDir=0;}
    if(shP&&!lastSh&&player.dashCD<=0){
      player.dashing=true;player.dashT=13;player.dashCD=110;
      for(let i=0;i<6;i++) addP(player.x+8,player.y+16,'#38bdf8',-player.facing*(1+Math.random()*3),(Math.random()-0.5)*4,18);
    }
  }

  if(jP&&!lastJump&&(player.onGround||player.jumpBuf>0)){
    player.vy=-11.5;player.onGround=false;player.jumpBuf=0;
    burst(player.x+8,player.y+player.h,'#c4b5fd',6,3,16,2);
    sfxJump();
  }
  if(player.jumpBuf>0) player.jumpBuf--;

  if(zP&&!lastZ){
    const bx=player.x+(player.facing>0?player.w+2:- 6);
    magicBolts.push({x:bx,y:player.y+12,vx:player.facing*6,vy:-0.3,fromBoss:false,life:90});
    burst(bx,player.y+12,'#a78bfa',6,3,18);
    player.shooting=12;
    sfxBlast();
  }

  player.vy+=0.47;
  player.x+=player.vx; player.y+=player.vy;
  player.x=Math.max(0,Math.min(W-player.w,player.x));

  if(player.y>H+80){player.x=LEVELS[cLvl].sx;player.y=LEVELS[cLvl].sy;player.vy=0;hurtPlayer(25);}

  player.onGround=false;
  for(const p of LEVELS[cLvl].plats){
    let py=p.y;
    const k=p.x+'|'+p.y;
    if(p.crumble&&crumbleState[k]) py+=crumbleState[k]*3;
    if(player.x+player.w>p.x&&player.x<p.x+p.w&&player.y+player.h>py&&player.y+player.h<py+p.h+Math.abs(player.vy)+2&&player.vy>=0){
      player.y=py-player.h; player.vy=0; player.onGround=true; player.jumpBuf=8;
      if(p.crumble){crumbleState[k]=(crumbleState[k]||0)+0.09;}
    }
  }

  lastJump=jP; lastZ=zP; lastSh=shP;
}

// ── COINS + PORTAL ───────────────────────────────────────────────────────────
let levelLocked=false;
function updateCoinsPortal(){
  portalAngle+=0.05;
  for(const c of coins){
    if(c.collected) continue;
    c.bob+=0.045;
    const by=Math.sin(c.bob)*3;
    if(overlap(player.x,player.y,player.w,player.h,c.x-7,c.y+by-7,14,14)){
      c.collected=true; score+=10;
      burst(c.x,c.y,'#fbbf24',10,5,24,3);
      sfxCoin();
    }
  }
  const lv=LEVELS[cLvl];
  if(!lv.boss&&!levelLocked&&coins.every(c=>c.collected)){
    // Portal center is at px+22, py-28 (doorway opening). Use box overlap.
    const portalCX=lv.px+22, portalCY=lv.py-28;
    const pdx=Math.abs(player.x+8-portalCX), pdy=Math.abs(player.y+16-portalCY);
    if(pdx<30&&pdy<40){
      levelLocked=true;
      score+=60;
      const nextLvl=cLvl+1;
      showMsg('<span class="title">Level '+(cLvl+1)+' Clear!</span>+60 bonus!<span class="sub">Loading next level...</span>',115);
      setTimeout(()=>{
        if(nextLvl<=14){cLvl=nextLvl;initLevel(cLvl);levelLocked=false;autoSave();}
        else{showMsg('<span class="title">YOU WIN!</span>Final Score: '+score,9999);}
      },2000);
    }
  }
}

function updateParticles(){
  for(let i=particles.length-1;i>=0;i--){
    const p=particles[i]; p.x+=p.vx; p.y+=p.vy; p.vy+=0.06; p.life--;
    if(p.life<=0) particles.splice(i,1);
  }
}
